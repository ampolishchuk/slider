import { ModelInteface, ModelStateType } from "./interfaces/Model.interface";
import ObserverInterface from "./interfaces/Observer.interface";
 
interface DependenciesInterface {
  state: ModelStateType;
  observer: ObserverInterface;
}
 
export default class Model implements ModelInteface {
  private observer: ObserverInterface;
  private state: ModelStateType;
 
  constructor(dependencies: DependenciesInterface) {
    const { observer, state } = dependencies;
 
    this.observer = observer;
    this.state = {
      ...state,
      width: state.container.offsetWidth,
      coords: state.container.getBoundingClientRect(),
    };
  }
 
  public getContainer(): HTMLElement {
    return this.state.container;
  }
 
  public setValue(value: any) {
    value = this.verifyValue(value);
 
    this.state.value = value;
 
    this.observer.notify("model:value", this.state.value);
  }
 
  public updateValue(index: number, clientX: number) {
    const value = this.calcValue(index, clientX);
 
    this.setValue(value);
  }
 
  public adjustPosition() {
    const { width, points, value } = this.state;
    const step = width / (points.length - 1);
    const valueDistance = points[points.length - 1] - points[0];
 
    for (let index in value) {
      // const position = step * points.indexOf(value[index]);
      const position = (width * value[index]) / valueDistance;
 
      this.observer.notify("model:position", { index, position });
    }
  }
 
  private calcValue(index: number, clientX: number): any {
    const { width, coords, points, value } = this.state;
    const ratio = (clientX - coords.left) / width;
 
    // const pointIndex = Math.round(ratio * (points.length - 1));
    // value[index] = points[pointIndex];
    const valueDistance = points[points.length - 1] - points[0];
    value[index] = valueDistance * ratio;
 
    return value;
  }
 
  private verifyValue(value: any): any {
    const { points } = this.state;
 
    if (!Array.isArray(value)) {
      const array = [value, points[0]];
 
      value = array;
    } else if (value.length != 2) {
      value[1] = points[0];
    }
 
    // value = value.map((pointValue: any) =>
    //   points.indexOf(pointValue) > 0 ? pointValue : points[0]
    // );
 
    return value;
  }
}
